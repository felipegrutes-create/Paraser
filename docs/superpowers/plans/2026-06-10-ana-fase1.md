```javascript
var CLAUDE_API_URL = 'https://api.anthropic.com/v1/messages';
var CLAUDE_MODEL   = 'claude-sonnet-4-6';

function _claudeKey_() {
  var k = PropertiesService.getScriptProperties().getProperty('ANTHROPIC_API_KEY');
  if (!k) throw new Error('ANTHROPIC_API_KEY não configurada');
  return k;
}

/**
 * Chama Claude com prompt simples.
 * @param {string} userMsg - mensagem do usuário
 * @param {string} systemPrompt - prompt do sistema (persona + regras)
 * @param {Array} historico - [{role:'user'|'assistant', content:'...'}, ...]
 * @return {string} texto da resposta de Claude
 */
function chamarClaude(userMsg, systemPrompt, historico) {
  var messages = (historico || []).slice();
  messages.push({ role: 'user', content: userMsg });

  var resp = UrlFetchApp.fetch(CLAUDE_API_URL, {
    method: 'post',
    contentType: 'application/json',
    headers: {
      'x-api-key': _claudeKey_(),
      'anthropic-version': '2023-06-01'
    },
    payload: JSON.stringify({
      model: CLAUDE_MODEL,
      max_tokens: 1024,
      system: systemPrompt || '',
      messages: messages
    }),
    muteHttpExceptions: true
  });

  var code = resp.getResponseCode();
  var body = resp.getContentText();
  if (code < 200 || code >= 300) {
    throw new Error('Claude HTTP ' + code + ': ' + body.substring(0, 300));
  }
  var json = JSON.parse(body);
  if (!json.content || !json.content[0] || !json.content[0].text) {
    throw new Error('Claude resposta inesperada: ' + body.substring(0, 300));
  }
  return json.content[0].text.trim();
}
```

```javascript
// ANA Bot — handler principal
function doGet(e) {
  var params = (e && e.parameter) || {};
  if (params.action === 'diag' && params.key === 'paraser2026') {
    return ContentService
      .createTextOutput(JSON.stringify(_diag(), null, 2))
      .setMimeType(ContentService.MimeType.JSON);
  }
  return ContentService.createTextOutput('ANA Bot online').setMimeType(ContentService.MimeType.TEXT);
}

function doPost(e) {
  try {
    var body = JSON.parse(e.postData.contents);
    if (body.fromMe === true) return _ok();

    var phone = body.phone || '';
    var texto = '';
    if (body.text && body.text.message) texto = String(body.text.message);
    else if (body.buttonsResponseMessage) texto = String(body.buttonsResponseMessage.message || body.buttonsResponseMessage.buttonId || '');

    if (!phone || !texto) return _ok();

    var resposta = _processarMensagem(phone, texto);
    if (resposta) _enviarZapi(phone, resposta);
  } catch (err) {
    Logger.log('doPost erro: ' + err.message + '\n' + (err.stack || ''));
    _logErro('doPost', err.message);
  }
  return _ok();
}

function _ok() {
  return ContentService.createTextOutput(JSON.stringify({ ok: true })).setMimeType(ContentService.MimeType.JSON);
}

function _processarMensagem(phone, texto) {
  // Hello world: ignora memória/FAQ por enquanto. Apenas responde com Claude usando system prompt fixo.
  var cfg = _lerConfig();
  var systemPrompt = 'Você é ' + cfg.persona_nome + ', atendente do Instituto Paraser.\n' +
                     'Tom: ' + cfg.persona_tom + '\n' +
                     'Responda BREVEMENTE (máx 3 frases) à mensagem do paciente.';
  return chamarClaude(texto, systemPrompt, []);
}

function _lerConfig() {
  var sh = _ss().getSheetByName('ANA_Config');
  var vals = sh.getRange(2, 1, sh.getLastRow() - 1, 2).getValues();
  var cfg = {};
  vals.forEach(function(r) { if (r[0]) cfg[r[0]] = r[1]; });
  return cfg;
}

function _enviarZapi(phone, msg) {
  var props = PropertiesService.getScriptProperties();
  var instanceId = props.getProperty('ZAPI_INSTANCE_ID');
  var token      = props.getProperty('ZAPI_TOKEN');
  var clientTok  = props.getProperty('ZAPI_CLIENT_TOKEN');
  if (!instanceId || !token) throw new Error('Z-API não configurado em Properties');

  var url = 'https://api.z-api.io/instances/' + instanceId + '/token/' + token + '/send-text';
  var headers = { 'Content-Type': 'application/json' };
  if (clientTok) headers['Client-Token'] = clientTok;

  UrlFetchApp.fetch(url, {
    method: 'post',
    headers: headers,
    payload: JSON.stringify({ phone: phone, message: msg }),
    muteHttpExceptions: true
  });
}

function _logErro(tipo, msg) {
  try {
    _ss().getSheetByName('ANA_Logs').appendRow([new Date(), tipo, 'ERRO', msg, '']);
  } catch (e) { /* fail silently se logs não der */ }
}

function _diag() {
  return {
    timestamp: new Date().toISOString(),
    has_anthropic_key: !!PropertiesService.getScriptProperties().getProperty('ANTHROPIC_API_KEY'),
    has_zapi: !!PropertiesService.getScriptProperties().getProperty('ZAPI_INSTANCE_ID'),
    has_sheet_id: !!PropertiesService.getScriptProperties().getProperty('SHEET_ID')

}
```

```javascript
var MEMORIA_MAX_MSGS = 10;

function memoriaCarregar(phone) {
  var row = _memoriaLinha(phone);
  if (!row) return [];
  try { return JSON.parse(row.mensagens_json || '[]'); } catch (e) { return []; }
}

function memoriaAppend(phone, role, content) {
  var hist = memoriaCarregar(phone);
  hist.push({ role: role, content: content, ts: new Date().toISOString() });
  while (hist.length > MEMORIA_MAX_MSGS) hist.shift();
  _memoriaSalvar(phone, hist);
}

function memoriaResetar(phone) {
  _memoriaSalvar(phone, [], 'RESETADO');
}
```

```javascript
var FEEGOW_BASE = 'https://api.feegow.com/v1/api';

function _feegowToken() {
  var t = PropertiesService.getScriptProperties().getProperty('FEEGOW_TOKEN');
  if (!t) throw new Error('FEEGOW_TOKEN não configurado');
  return t;
}

/**
 * Busca agendamentos do dia + extrai gaps (slots livres) por médico.
 * @param {string} nomeMedico - "Rodolfo", "Marcelle", etc.
 * @param {Date} data - dia desejado
 * @return {Array} [{ hora: '14:00', medico: 'RODOLFO SALVATO', profId: 3 }]
 */
function buscarDisponibilidade(nomeMedico, data) {
  var ds = Utilities.formatDate(data, 'GMT-3', 'yyyy-MM-dd');
  var url = FEEGOW_BASE + '/appoints/search?data_start=' + ds + '&data_end=' + ds;
  var resp = UrlFetchApp.fetch(url, {
    headers: { 'x-access-token': _feegowToken() },
    muteHttpExceptions: true
  });
  var json = JSON.parse(resp.getContentText());
  var ags = (json.content || []).filter(function(a) {
    // exclui cancelados (status 11/15/22) e libera horários
    var st = Number(a.status_id);
    return st !== 11 && st !== 15 && st !== 22;
  });

  // Carrega mapping profId → nome
  var profMap = _profissionaisMap();
  var ocupados = {};
  var profIdAlvo = null;
  ags.forEach(function(a) {
    var nome = (profMap[a.profissional_id] || '').toUpperCase();
    if (!profIdAlvo && nome.indexOf(String(nomeMedico).toUpperCase()) !== -1) profIdAlvo = a.profissional_id;
    if (nome.indexOf(String(nomeMedico).toUpperCase()) !== -1) {
      ocupados[a.horario] = true;
    }
  });

  if (!profIdAlvo) return [];

  // Gera horários candidatos 8h-18h de 30 em 30min e remove ocupados
  var livres = [];
  for (var h = 8; h < 18; h++) {
    for (var m of [0, 30]) {
      var hh = (h < 10 ? '0' + h : h) + ':' + (m === 0 ? '00' : '30') + ':00';
      if (!ocupados[hh]) {
        livres.push({ hora: hh.substring(0, 5), medico: profMap[profIdAlvo], profId: profIdAlvo });
      }
    }
  }
  return livres;
}

function _profissionaisMap() {
  var cache = CacheService.getScriptCache();
  var cached = cache.get('PROFMAP');
  if (cached) return JSON.parse(cached);

  var resp = UrlFetchApp.fetch(FEEGOW_BASE + '/professional/list', {
    headers: { 'x-access-token': _feegowToken() },
    muteHttpExceptions: true
  });
  var json = JSON.parse(resp.getContentText());
  var map = {};
  (json.content || []).forEach(function(p) { map[p.id] = (p.nome || '').toUpperCase(); });
  cache.put('PROFMAP', JSON.stringify(map), 3600);
  return map;
}
```

```javascript
function _processarMensagem(phone, texto) {
  var cfg = _lerConfig();
  var historico = memoriaCarregar(phone);

  var rf = detectarRedFlag(texto);
  if (rf) {
    memoriaAppend(phone, 'user', texto);
    var resp = respostaRedFlag(rf);
    memoriaAppend(phone, 'assistant', resp);
    var row = _memoriaLinha(phone);
    escalarRedFlag(phone, row ? row.nome : '', rf, texto);
    return resp;
  }

  if (historico.length === 0) {
    memoriaAppend(phone, 'user', texto);
    memoriaAppend(phone, 'assistant', cfg.boas_vindas);
    return cfg.boas_vindas;
  }

  var systemPrompt = _montarSystemPrompt(cfg);
  memoriaAppend(phone, 'user', texto);
  var hist = memoriaCarregar(phone);
  var ctx = hist.slice(0, -1).map(function(m) { return { role: m.role, content: m.content }; });
  var bruta = chamarClaude(texto, systemPrompt, ctx);

  // Parse marcação de pré-agendamento
  var match = bruta.match(/\[\[PRE_AGENDAR:\s*medico\s*=\s*"([^"]+)"\s*,\s*quando\s*=\s*"([^"]+)"(?:\s*,\s*procedimento\s*=\s*"([^"]+)")?\s*\]\]/);
  if (match) {
    try {
      var dataHora = _parsearQuando(match[2]);
      var rowMem = _memoriaLinha(phone);
      registrarPreAgendamento(phone, rowMem ? rowMem.nome : '', match[1], dataHora, match[3] || 'consulta');
      memoriaSetStatus(phone, 'PRE_AGENDADO');
    } catch (e) {
      Logger.log('Falha parsing pré-agendamento: ' + e.message);
    }
  }

  var limpa = bruta.replace(/\[\[PRE_AGENDAR:[^\]]+\]\]/g, '').trim();
  memoriaAppend(phone, 'assistant', limpa);
  return limpa;
}

function _parsearQuando(s) {
  // aceita "16/06 14:00", "16/06 14h", "16/06 às 14:00", etc.
  var m = String(s).match(/(\d{1,2})\/(\d{1,2})\s*(?:às\s*)?(\d{1,2})[:h]?(\d{0,2})/i);
  if (!m) throw new Error('formato data inválido: ' + s);
  var ano = new Date().getFullYear();
  var d = new Date(ano, +m[2] - 1, +m[1], +m[3], +(m[4] || 0));
  if (d < new Date()) d.setFullYear(ano + 1); // se passou, é ano que vem
  return d;
}
```
